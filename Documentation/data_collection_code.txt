#include <ModbusMaster.h>
#include <OneWire.h>
#include <DallasTemperature.h>

// ==========================================
// PIN DEFINITIONS
// ==========================================
#define MAX485_DE_RE 23
#define ONE_WIRE_BUS 4
#define MOISTURE_PIN 34

// CHANGE THIS FOR EACH POT
//#define DATA_LABEL "Depleted"
//#define DATA_LABEL "Moderate"
#define DATA_LABEL "Nutrient-Rich"

// ==========================================
// MOISTURE CALIBRATION
// ==========================================
#define DRY_VALUE 2750
#define WET_VALUE 1100

// ==========================================
// SAMPLING INTERVAL
// ==========================================
#define SAMPLE_INTERVAL_MS 30000

// ==========================================
// OBJECTS
// ==========================================
ModbusMaster node;
OneWire oneWire(ONE_WIRE_BUS);
DallasTemperature tempSensor(&oneWire);

// ==========================================
// VARIABLES
// ==========================================
int nitrogen = 0;
int phosphorus = 0;
int potassium = 0;

float temperature = 0.0;
float moisture = 0.0;

bool npkValid = false;

unsigned long lastSampleTime = 0;

// ==========================================
// MAX485 CONTROL
// ==========================================
void preTransmission() {
  digitalWrite(MAX485_DE_RE, HIGH);
  delayMicroseconds(100);
}

void postTransmission() {
  delayMicroseconds(100);
  digitalWrite(MAX485_DE_RE, LOW);
}

// ==========================================
// MOISTURE PERCENT
// ==========================================
float readMoisturePercent() {

  int raw = analogRead(MOISTURE_PIN);

  float pct = map(
    raw,
    DRY_VALUE,
    WET_VALUE,
    0,
    100
  );

  return constrain(pct, 0.0, 100.0);
}

// ==========================================
// SETUP
// ==========================================
void setup() {

  Serial.begin(115200);
  delay(1000);

  // MAX485
  pinMode(MAX485_DE_RE, OUTPUT);
  digitalWrite(MAX485_DE_RE, LOW);

  // NPK SENSOR
  Serial1.begin(9600, SERIAL_8N1, 16, 17);

  node.begin(1, Serial1);
  node.preTransmission(preTransmission);
  node.postTransmission(postTransmission);

  // TEMPERATURE
  tempSensor.begin();

  // MOISTURE
  pinMode(MOISTURE_PIN, INPUT);
  analogReadResolution(12);

  Serial.println("Timestamp,N,P,K,Moisture,Temperature,Label");
}

// ==========================================
// LOOP
// ==========================================
void loop() {

  if (millis() - lastSampleTime >= SAMPLE_INTERVAL_MS) {

    lastSampleTime = millis();

    // --------------------------------------
    // READ NPK
    // --------------------------------------
    uint8_t result = node.readHoldingRegisters(0x001E, 3);

//    Serial.print("Modbus Result = ");
//    Serial.println(result);

    if (result == node.ku8MBSuccess) {

      nitrogen   = node.getResponseBuffer(0);
      phosphorus = node.getResponseBuffer(1);
      potassium  = node.getResponseBuffer(2);

      npkValid = true;

    } else {

      npkValid = false;
      return;
    }

    // --------------------------------------
    // READ TEMPERATURE
    // --------------------------------------
    tempSensor.requestTemperatures();
    temperature = tempSensor.getTempCByIndex(0);

    if (temperature == DEVICE_DISCONNECTED_C) {
      return;
    }

    // --------------------------------------
    // READ MOISTURE
    // --------------------------------------
    int raw = analogRead(MOISTURE_PIN);

//   Serial.print("RAW=");
//   Serial.println(raw);
    moisture = readMoisturePercent();

    // --------------------------------------
    // CSV OUTPUT
    // --------------------------------------
    Serial.print(millis());
    Serial.print(",");

    Serial.print(nitrogen);
    Serial.print(",");

    Serial.print(phosphorus);
    Serial.print(",");

    Serial.print(potassium);
    Serial.print(",");

    Serial.print(moisture, 1);
    Serial.print(",");

    Serial.print(temperature, 2);
    Serial.print(",");

    Serial.println(DATA_LABEL);
  }
}
