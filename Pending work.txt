Step 2 — Train Random Forest Model (Implemented in train_pipeline.py)
pythonfrom sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import classification_report, confusion_matrix
import seaborn as sns
import matplotlib.pyplot as plt

# Scale features
scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)

# Save scaler parameters for ESP32 firmware
print("Scaler means:", scaler.mean_)
print("Scaler stds:", scaler.scale_)

# Train test split
X_train, X_test, y_train, y_test = train_test_split(
    X_scaled, y, test_size=0.2, stratify=y, random_state=42)

# Train
model = RandomForestClassifier(n_estimators=50, random_state=42)
model.fit(X_train, y_train)

# Evaluate
y_pred = model.predict(X_test)
print(classification_report(y_test, y_pred))

# Confusion matrix
cm = confusion_matrix(y_test, y_pred)
sns.heatmap(cm, annot=True, fmt='d',
            xticklabels=model.classes_,
            yticklabels=model.classes_)
plt.title('Confusion Matrix')
plt.show()

# Feature importance
feat_imp = pd.Series(model.feature_importances_,
                     index=['N','P','K','Moisture','Temperature'])
feat_imp.sort_values().plot(kind='barh')
plt.title('Feature Importance')
plt.show()

Step 4 — Integrate Model into ESP32 Firmware
Paste the exported C function into a new file model.h in your Arduino project. In your main firmware:
cpp// In model.h — paste m2cgen output here
double score(double * input);  // function signature

// In main firmware
#include "model.h"

String runInference(int n, int p, int k,
                    float moisture, float temp) {
  // Apply StandardScaler using hardcoded parameters
  double input[5];
  input[0] = (n        - N_mean)         / N_std;
  input[1] = (p        - P_mean)         / P_std;
  input[2] = (k        - K_mean)         / K_std;
  input[3] = (moisture - Moisture_mean)  / Moisture_std;
  input[4] = (temp     - Temp_mean)      / Temp_std;

  // Run inference
  double result[3];  // 3 classes
  score(input, result);

  // Find class with highest score
  int maxIdx = 0;
  for (int i = 1; i < 3; i++) {
    if (result[i] > result[maxIdx]) maxIdx = i;
  }

  // Map index to label
  if (maxIdx == 0) return "Depleted";
  if (maxIdx == 1) return "Moderate";
  return "Nutrient-Rich";
}

Phase 3 — MQTT Communication (✅ Completed and Verified)
What MQTT Is
MQTT (Message Queuing Telemetry Transport) is a lightweight publish-subscribe protocol designed specifically for IoT devices. It works like this:
ESP32 (Publisher)
      ↓  publishes message to topic
Mosquitto Broker (running on laptop)
      ↓  forwards to all subscribers
Node-RED (Subscriber)
      ↓  processes and stores
InfluxDB → Grafana (Dashboard)
Three components:

Publisher — ESP32 sends data
Broker — Mosquitto receives and routes messages
Subscriber — Node-RED receives and processes

MQTT Topic Structure
Your ESP32 publishes to three topics:
TopicContentTriggersoil/readingsN, P, K, moisture, temp, classEvery 30 secondssoil/alertsCUSUM score, drift alertWhen drift detectedsoil/fertilizationEvent timestamp, effectivenessAfter 48h window
ESP32 MQTT Code Addition
Add this to your existing firmware:
cpp#include <WiFi.h>
#include <PubSubClient.h>
#include <ArduinoJson.h>

// ── WiFi credentials ──
const char* WIFI_SSID     = "YourWiFiName";
const char* WIFI_PASSWORD = "YourWiFiPassword";

// ── MQTT broker ──
// Use your laptop's IP address on the same WiFi network
const char* MQTT_BROKER   = "192.168.1.100";
const int   MQTT_PORT     = 1883;
const char* MQTT_CLIENT   = "ESP32_SoilMonitor";

WiFiClient espClient;
PubSubClient mqtt(espClient);

void connectWiFi() {
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
  Serial.print("Connecting to WiFi");
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nWiFi connected: " + WiFi.localIP().toString());
}

void connectMQTT() {
  while (!mqtt.connected()) {
    if (mqtt.connect(MQTT_CLIENT)) {
      Serial.println("MQTT connected");
    } else {
      Serial.print("MQTT failed, rc=");
      Serial.println(mqtt.state());
      delay(2000);
    }
  }
}

void publishReadings(int n, int p, int k,
                     float moisture, float temp,
                     String fertilityClass,
                     float cusumScore, bool driftAlert) {
  StaticJsonDocument<256> doc;
  doc["N"]          = n;
  doc["P"]          = p;
  doc["K"]          = k;
  doc["moisture"]   = moisture;
  doc["temp"]       = temp;
  doc["class"]      = fertilityClass;
  doc["cusum"]      = cusumScore;
  doc["drift"]      = driftAlert;
  doc["timestamp"]  = millis();

  char payload[256];
  serializeJson(doc, payload);
  mqtt.publish("soil/readings", payload);
}

void publishAlert(float cusumScore) {
  StaticJsonDocument<128> doc;
  doc["type"]       = "DEPLETION_DRIFT";
  doc["cusum"]      = cusumScore;
  doc["timestamp"]  = millis();

  char payload[128];
  serializeJson(doc, payload);
  mqtt.publish("soil/alerts", payload);
}
Setting Up Mosquitto Broker on Laptop
Windows:
1. Download Mosquitto from mosquitto.org
2. Install it
3. Open Command Prompt as Administrator
4. Run: net start mosquitto
5. Verify: mosquitto -v
Find your laptop IP address:
Windows: ipconfig → look for IPv4 Address
Mac/Linux: ifconfig → look for inet address
Use this IP as MQTT_BROKER in your ESP32 code. Both ESP32 and laptop must be on the same WiFi network.
